#float-img {
	position: fixed;
	right: 5px;
	bottom: 5px;
    _position: absolute;
    _zoom: expression(function(el) {
              el.style.zoom = 1;

              var docEl = document.documentElement;
              window.onscroll = function() {
                  el.style.top = (docEl.scrollTop + docEl.clientHeight - el.clientHeight - 5) + "px";
              };
          }(this));
}
